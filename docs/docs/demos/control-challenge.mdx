  solution={`import math

def body_to_earth(u, v, w, phi, theta, psi):
    """
    机体坐标系 -> 地理坐标系 (NED) 速度变换
    R = Rz(psi) @ Ry(theta) @ Rx(phi)
    """
    cp, sp = math.cos(phi), math.sin(phi)
    ct, st = math.cos(theta), math.sin(theta)
    cy, sy = math.cos(psi), math.sin(psi)

    # 旋转矩阵 R = Rz @ Ry @ Rx
    r00 = cy * ct
    r01 = cy * st * sp - sy * cp
    r02 = cy * st * cp + sy * sp

    r11 = sy * st * sp + cy * cp
    r12 = sy * st * cp - cy * sp

    r21 = ct * sp
    r22 = ct * cp

    vn = r00 * u + r01 * v + r02 * w
    ve = r10 * u + r11 * v + r12 * w
    vd = r20 * u + r21 * v + r22 * w

    return vn, ve, vd

vn, ve, vd = body_to_earth(20, 0, -2, math.radians(10), math.radians(5), math.radians(30))

    前向欧拉法: v(n+1) = v(n) + a*dt, p(n+1) = p(n) + v(n)*dt

  solution={`def rk4_step(f, t, y, dt):
    """
    四阶龙格-库塔法 (RK4)

    精度: O(dt⁵)，比欧拉法 O(dt²) 高四个数量级
    """
    k1 = f(t, y)
    k2 = f(t + dt / 2, [yi + dt / 2 * ki for yi, ki in zip(y, k1)])
    k3 = f(t + dt / 2, [yi + dt / 2 * ki for yi, ki in zip(y, k2)])
